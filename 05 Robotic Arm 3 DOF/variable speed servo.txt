#include <VarSpeedServo.h> 

// create servo object to control a servo 
VarSpeedServo myservo1;
 
void setup() {

  // initialize serial:
 //Serial.begin(9600);
  
  myservo1.attach(6);
  
} 
 
void loop() {
  
  int LEF = 0;
  int RIG = 180;
  
  int SPEED1 = 10;
  int SPEED2 =100;
 
  
  for(int i = 0; i < 2; i++) {

    myservo1.write(LEF, SPEED1);    
    myservo1.wait();
      
    myservo1.write(RIG, SPEED2);     
    myservo1.wait();
    
    myservo1.write(LEF, SPEED1); 
     
    myservo1.wait();
              
    myservo1.write(RIG, SPEED2);     
    myservo1.wait();
    
  }
  
  delay(2000);
}
